generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==========================================
// USERS & PERMISSIONS
// ==========================================

model User {
  id               String           @id @default(uuid())
  email            String           @unique
  passwordHash     String
  resetToken       String?          // Added for Sakshi's review
  resetTokenExpiry DateTime?        // Added for Sakshi's review
  roleId           String
  role             Role             @relation(fields: [roleId], references: [id])
  employee         Employee?
  authSessions     Authentication[] // Added for Sakshi's review
  createdAt        DateTime         @default(now())
  updatedAt        DateTime         @updatedAt
}

// Authentication entity – Added for Sakshi's review
// Tracks active sessions, refresh tokens, and login history
model Authentication {
  id           String   @id @default(uuid())
  userId       String
  user         User     @relation(fields: [userId], references: [id])
  token        String   @unique // JWT refresh token or session token
  deviceInfo   String?  // e.g. "Chrome on Windows"
  ipAddress    String?
  isActive     Boolean  @default(true)
  expiresAt    DateTime
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}

// Normalized from Enum
model Role {
  id          String       @id @default(uuid())
  name        String       @unique // e.g. SUPER_ADMIN, HR_MANAGER, SALES_MANAGER
  description String?
  users       User[]
  permissions Permission[] // Mapping to permissions
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt
}

// New table to handle module-level access
model Permission {
  id          String   @id @default(uuid())
  module      Module
  action      Action
  roleId      String
  role        Role     @relation(fields: [roleId], references: [id])
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

enum Module {
  HR
  CRM
  INVENTORY
  PROJECTS
}

enum Action {
  READ
  WRITE
  DELETE
  ALL
}


// ==========================================
// HR MODULE
// ==========================================

model Employee {
  id            String       @id @default(uuid())
  userId        String?      @unique
  user          User?        @relation(fields: [userId], references: [id])
  firstName     String
  lastName      String
  gender        String?
  dob           DateTime?
  contact       String?
  address       String?
  city          String?
  state         String?
  country       String?
  joinDate      DateTime     @default(now())
  empType       String       // Full-time, Part-time, Contract
  status        String       // Active, Inactive
  
  departmentId  String
  department    Department   @relation(fields: [departmentId], references: [id])
  
  designationId String
  designation   Designation  @relation(fields: [designationId], references: [id])

  attendances   Attendance[] // Added for Sakshi's review
  leaves        Leave[]      // Added for Sakshi's review
  assignments   Assignment[] // Added for Sakshi's review
  performanceReviews PerformanceReview[] // Added for Sakshi's review
  
  createdAt     DateTime     @default(now())
  updatedAt     DateTime     @updatedAt
}

// Normalized from plain text
model Department {
  id          String     @id @default(uuid())
  name        String     @unique // e.g. Engineering, Sales, HR
  description String?
  employees   Employee[]
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt
}

// Normalized from plain text
model Designation {
  id          String     @id @default(uuid())
  title       String     @unique // e.g. Senior Developer, HR Manager
  description String?
  employees   Employee[]
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt
}

// Added for Sakshi's review
model Attendance {
  id         String   @id @default(uuid())
  employeeId String
  employee   Employee @relation(fields: [employeeId], references: [id])
  date       DateTime
  status     String   // Present, Absent, Half Day, Late
  checkIn    DateTime?
  checkOut   DateTime?
  hoursWorked Float?
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
}

// Added for Sakshi's review
model Leave {
  id         String   @id @default(uuid())
  employeeId String
  employee   Employee @relation(fields: [employeeId], references: [id])
  leaveType  String   // Sick Leave, Casual Leave, Annual Leave
  startDate  DateTime
  endDate    DateTime
  status     String   // Pending, Approved, Rejected
  reason     String   // Enforced required for frontend alignment
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
}


// ==========================================
// CRM MODULE
// ==========================================

model Lead {
  id        String      @id @default(uuid())
  name      String
  email     String?
  phone     String?
  company   String?
  status    String      // New, Contacted, Qualified, Converted, Lost
  customer  Customer?   // Added for Sakshi's review (converted relation)
  followUps FollowUp[]  // Added for Sakshi's review
  createdAt DateTime    @default(now())
  updatedAt DateTime    @updatedAt
}

// Added for Sakshi's review
model Customer {
  id                 String        @id @default(uuid())
  name               String
  email              String?
  phone              String?
  company            String?
  convertedFromLeadId String?      @unique
  lead               Lead?         @relation(fields: [convertedFromLeadId], references: [id])
  opportunities      Opportunity[]
  followUps          FollowUp[]
  createdAt          DateTime      @default(now())
  updatedAt          DateTime      @updatedAt
}

// Added for Sakshi's review
model Opportunity {
  id                String   @id @default(uuid())
  customerId        String
  customer          Customer @relation(fields: [customerId], references: [id])
  value             Float
  stage             String   // Discovery, Proposal, Negotiation, Closed Won, Closed Lost
  expectedCloseDate DateTime?
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt
}

// Added for Sakshi's review
/// @validation: Enforce that at least one of leadId or customerId is provided
model FollowUp {
  id            String    @id @default(uuid())
  leadId        String?
  lead          Lead?     @relation(fields: [leadId], references: [id])
  customerId    String?
  customer      Customer? @relation(fields: [customerId], references: [id])
  date          DateTime
  notes         String
  nextActionDate DateTime?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
}


// ==========================================
// INVENTORY MODULE
// ==========================================

model Product {
  id             String          @id @default(uuid())
  name           String
  sku            String          @unique
  category       String
  price          Float
  costPrice      Float?
  stockLevel     Int             @default(0)
  minStockLevel  Int             @default(0)
  unit           String          // Piece, Kg, Box
  status         String          // Active, Inactive
  purchaseOrders PurchaseOrder[] // Added for Sakshi's review
  stockMovements StockMovement[] // Added for Sakshi's review
  createdAt      DateTime        @default(now())
  updatedAt      DateTime        @updatedAt
}

// Supplier entity – Added for Sakshi's review
model Supplier {
  id             String          @id @default(uuid())
  name           String
  contactPerson  String?
  email          String?
  phone          String?
  address        String?
  city           String?
  country        String?
  status         String          @default("Active") // Active, Inactive
  purchaseOrders PurchaseOrder[]
  createdAt      DateTime        @default(now())
  updatedAt      DateTime        @updatedAt
}

// Added for Sakshi's review
model PurchaseOrder {
  id         String   @id @default(uuid())
  supplierId String
  supplier   Supplier @relation(fields: [supplierId], references: [id]) // Now a proper FK relation
  productId  String
  product    Product  @relation(fields: [productId], references: [id])
  quantity   Int
  orderDate  DateTime
  status     String   // Ordered, Shipped, Delivered, Cancelled
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
}

// Added for Sakshi's review
model StockMovement {
  id           String   @id @default(uuid())
  productId    String
  product      Product  @relation(fields: [productId], references: [id])
  changeAmount Int      // Positive for restock, negative for sale/loss
  reason       String   // Sale, Restock, Adjustment, Return
  date         DateTime // Set independently to represent actual movement date
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}


// ==========================================
// PROJECTS MODULE
// ==========================================

model Project {
  id          String      @id @default(uuid())
  name        String
  description String?
  status      String      // Not Started, In Progress, On Hold, Completed
  priority    String      // Low, Medium, High, Critical
  startDate   DateTime?
  endDate     DateTime?
  tasks       Task[]
  milestones  Milestone[] // Added for Sakshi's review
  assignments Assignment[] // Added for Sakshi's review
  createdAt   DateTime    @default(now())
  updatedAt   DateTime    @updatedAt
}

model Task {
  id          String       @id @default(uuid())
  projectId   String
  project     Project      @relation(fields: [projectId], references: [id])
  title       String
  description String?
  status      String       // Not Started, In Progress, Review, Done
  priority    String       // Low, Medium, High, Critical
  dueDate     DateTime?
  assignments Assignment[] // Added for Sakshi's review
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt
}

// Assignment entity – Added for Sakshi's review
// Tracks which employee is assigned to which task/project
model Assignment {
  id         String    @id @default(uuid())
  taskId     String?
  task       Task?     @relation(fields: [taskId], references: [id])
  projectId  String?
  project    Project?  @relation(fields: [projectId], references: [id])
  employeeId String
  employee   Employee  @relation(fields: [employeeId], references: [id])
  role       String?   // e.g. Lead Developer, Reviewer, Tester
  assignedAt DateTime  @default(now())
  createdAt  DateTime  @default(now())
  updatedAt  DateTime  @updatedAt
}

// Added for Sakshi's review
model Milestone {
  id        String   @id @default(uuid())
  projectId String
  project   Project  @relation(fields: [projectId], references: [id])
  title     String
  dueDate   DateTime?
  status    String   // Pending, Reached
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// Added for Sakshi's review
model PerformanceReview {
  id          String   @id @default(uuid())
  employeeId  String
  employee    Employee @relation(fields: [employeeId], references: [id])
  quarter     String   // e.g. Q1, Q2, Q3, Q4
  rating      Int      // 1-5
  review      String
  goals       String?  // Optional goals
  reviewer    String   // Reviewer name/role
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}
